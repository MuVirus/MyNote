
``` cpp
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
    ~TreeNode() {
        // 递归删除左子树：如果左子节点存在，先销毁它（会触发左子节点的析构）
        if (left != nullptr) {
            delete left;
            left = nullptr; // 避免野指针（可选，但更安全）
        }
        // 递归删除右子树：同理
        if (right != nullptr) {
            delete right;
            right = nullptr;
        }
        // 当前节点的内存会由delete操作自动释放
    }    
};
```
