# 二叉树
## 定义
### 简单二叉树结构（裸指针）
``` cpp
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
    ~TreeNode() {
        if (left != nullptr) {
            delete left;
            left = nullptr;
        }
        if (right != nullptr) {
            delete right;
            right = nullptr;
        }
    }    
};
```


## 遍历
### 前中后序遍历
**前序遍历**：根-左-右。先获取根节点值，再访问根的左子树，最后访问根的右子树。
**中序遍历**：左-根-右。先访问根的左子树，再获取根节点值，最后访问根的右子树。
**后序遍历**：左-右-根。先访问根的左子树，再访问根的右子树，最后获取根节点值。
### 递归法
#### 前/中/后序遍历
##### 通用模板（前序遍历为模板）
``` cpp
class Solution {
public:
    void traversal(TreeNode* node, vector<int>& vec) {
        if(node == nullptr) return;
        vec.push_back(node->val);//根据根的位置移动这一行即可。
        traversal(node->left, vec);
        traversal(node->right, vec);
    }
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> ans;
        traversal(root, ans);
        return ans;
    }
};
```
注意
- 递归写法最好单独放到一个函数中去写，因为可以按照自己需求定义函数的返回值/参数，避免一些内存浪费。
##### lambda写法（C++11）（中序遍历为例）
``` cpp
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> vec;
        function<void(TreeNode* node)> traversal = [&](TreeNode* node){
            if(node == nullptr) return;
            traversal(node->left);
            vec.push_back(node->val);
            traversal(node->right);
        };
        traversal(root);
        return vec;
    }
```

### 迭代法
#### 前/中序遍历
##### 通用模板
``` cpp
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> vec;
        stack<TreeNode*> st;
        while(root || !st.empty()) {
            if(root) {
                st.push(root);
                // vec.push_back(root->val); // 前序遍历（先根）
                root = root->left; // 当root有效时，优先遍历左节点
            } else {
                root = st.top();
                st.pop();
                // vec.push_back(root); // 中序遍历（先左）
                root = root->right; // 当root左子树遍历完时，再遍历右节点
            }
        }
        return vec;
    }
};
```
#### 后序遍历
后续遍历的顺序是：`左-右-根`，所以说我们只需要按照`根-右-左`的顺序进行遍历得到数组，最后翻转一下即可。
整体模板还是按照`先序遍历`来写，因为原本顺序为`根左右`，不过翻转之后为`右左根`，不符合要求，所以只需要修改一下为先遍历右节点即可。
``` cpp
class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> vec;
        stack<TreeNode*> st;
        while(root || !st.empty()) {
            if(root) {
                st.push(root);
                vec.push_back(root->val);
                root = root->right; // 当root有效时，优先遍历右节点
            } else {
                root = st.top();
                st.pop();
                root = root->left; // 当root右子树遍历完时，再遍历右节点
            }
        }
        reverse(vec.begin(), vec.end()); // 最后翻转
        return vec;
    }
};
```


## 题型
### 对称二叉树
以`左-右-根`的顺序来遍历，和`根-右-左`的顺序来遍历进行比较，在`递归`过程中做比较，因为过程中可以更简单的实现区分nullptr vs 非nullptr的情况。
``` cpp
class Solution {
public:
    bool compare(TreeNode* left, TreeNode* right) {
        if(left == nullptr && right == nullptr) return true;
        else if(left == nullptr && right)   return false;
        else if(left && right == nullptr)   return false;
        if(left->val != right->val) return false;
        auto outside = compare(left->left, right->right);
        auto inside = compare(left->right, right->left);
        return (outside && inside);
    }
    bool isSymmetric(TreeNode* root) {
        return compare(root->left, root->right);
    }
};
```

### 构造二叉树
#### 通过前序+中序遍历序列构造二叉树
##### 递归+哈希表
``` cpp
class Solution {
public:
    unordered_map<int, int> mp;
    // root：根 l：是否为左节点
    void process(vector<int>& preorder, vector<int>& inorder, int prel, int prer, int inl, int inr, TreeNode* root, bool l) {
        // 左/右子树为空，则返回
        if(prel > prer) return;
        // 如果不为空，则链接父节点
        TreeNode *node = new TreeNode(preorder[prel]);
        if(l) {
            root->left = node;
        } else {
            root->right = node;
        }
        // 找到根在中序遍历中的序号
        int inRoot = mp[preorder[prel]];
        
        // 左树 中序：左边界为inl，右边界为inRoot - 1
        process(preorder, inorder, prel + 1, prel + inRoot - inl, inl, inRoot - 1, node, true);
        // 右树 中序：左边界为inRoot + 1，右边界为inr
        process(preorder, inorder, prel + inRoot - inl + 1, prer, inRoot + 1, inr, node, false);
    }
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        if(preorder.empty() || inorder.empty()) return nullptr;
        TreeNode* dummyNode = new TreeNode();
        for(int i = 0; i < inorder.size(); ++i) {
            mp.emplace(inorder[i], i);
        }
        process(preorder, inorder, 0, preorder.size() - 1, 0, inorder.size() - 1, dummyNode, true);
        TreeNode *root = dummyNode->left;
        delete dummyNode;
        dummyNode = nullptr;
        return root;
    }
};
```

### 路径总和
#### 前缀和+哈希表+前序遍历（迭代法）
``` cpp
class Solution {
public:
    int count = 0;
    unordered_map<long long, int> mp;
    int target;
    void traversal(TreeNode* node, long long presum) {
        if(node == nullptr) return;
        // 前序遍历：先根
        presum += node->val;
        if(mp.find(presum - target) != mp.end()) {
            count += mp[presum - target];
        }
        ++mp[presum];
        traversal(node->left, presum);
        traversal(node->right, presum);
		// 递归遍历到叶子节点时会开始逐层往回走
		// 都要把这个节点对应的前缀和从哈希表里 “撤掉”
		// 这样后面遍历其他分支时，就不会用到这个节点的前缀和，避免算错路径数。
        --mp[presum];
        if(mp[presum] == 0) {
            mp.erase(presum);
        }
    }
    int pathSum(TreeNode* root, int targetSum) {
        mp[0] = 1;
        target = targetSum;
        traversal(root, 0);
        return count;
    }
};
```
